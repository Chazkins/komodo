SELECT
    з.id_заказа,
    з.номер_заказа,
    зк.наименование                          AS заказчик,
    п.наименование                           AS продукция,
    сз.количество                            AS кол_во_в_заказе,
    сз.цена_за_единицу                       AS цена_реализации,
    сз.сумма_строки                          AS выручка_по_строке,
    стоимость_матер.себестоимость_единицы    AS себестоимость_единицы,
    ROUND(сз.количество * стоимость_матер.себестоимость_единицы, 2) AS полная_стоимость_материалов

FROM заказ з
JOIN заказчик зк  ON зк.id_заказчика  = з.id_заказчика
JOIN строка_заказа сз ON сз.id_заказа = з.id_заказа
JOIN продукция п  ON п.id_продукции   = сз.id_продукции
JOIN (
    SELECT
        с.id_продукции,
        SUM(сс.количество_материала * м.цена_закупки) AS себестоимость_единицы
    FROM спецификация с
    JOIN строка_спецификации сс ON сс.id_спецификации = с.id_спецификации
    JOIN материал м             ON м.id_материала     = сс.id_материала
    GROUP BY с.id_продукции
) стоимость_матер ON стоимость_матер.id_продукции = сз.id_продукции

WHERE з.id_заказа = 1  -- <-- вот сюда подставляй нужный id

ORDER BY п.наименование;
